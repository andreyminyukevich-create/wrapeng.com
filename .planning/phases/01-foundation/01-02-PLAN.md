async function onDrop(e) {
  e.preventDefault();
  const col = e.currentTarget.closest('.column');
  col?.classList.remove('drag-over');
  const toStatus = col?.dataset.status;
  if (!toStatus || !_dragCalcId || toStatus === _dragStatus) return;

  // Transition guard — card snaps back naturally (DOM not modified)
  if (!isAllowed(_dragStatus, toStatus)) {
    showToast('warning', 'Переход недоступен');
    return;
  }

  const calc = allCalcs.find(c => c.id === _dragCalcId);
  const carName = calc?.car_name || 'Без названия';

  // Modal-requiring transitions
  if (toStatus === 'accepted') {
    PipelineForms.openModalAccept(_dragCalcId, carName);
    return;
  }
  if (toStatus === 'done') {
    openModalDone(_dragCalcId, carName);
    return;
  }
  if (toStatus === 'delivered') {
    openModalDeliver(_dragCalcId, carName, calc?.total_price || 0);
    return;
  }

  // Direct transitions (no modal needed)
  const ok = await updateStatus(_dragCalcId, toStatus);
  if (ok) {
    const label = COLUMNS.find(c => c.key === toStatus)?.label || toStatus;
    showToast('success', `→ ${label}`);
    loadBoard();
  }
}

```javascript
async function updateStatus(calcId, status, extra = {}) {
  // Strip internal metadata before sending to DB
  const historyComment = extra._historyComment || null;
  delete extra._historyComment;

  const { error } = await sb
    .from('calculations')
    .update({ status, updated_at: new Date().toISOString(), ...extra })
    .eq('id', calcId)
    .eq('studio_id', studioCtx.studioId);
  if (error) {
    console.error('[board] updateStatus:', error);
    showToast('error', 'Ошибка обновления статуса');
    return false;
  }
  // Record status history (BOARD-04) — fire-and-forget, non-blocking
  const fromStatus = allCalcs.find(c => c.id === calcId)?.status || null;
  sb.from('status_history').insert({
    calc_id: calcId,
    studio_id: studioCtx.studioId,
    from_status: fromStatus,
    to_status: status,
    changed_by: studioCtx.user?.id || null,
    comment: historyComment,
  }).then(({ error: hErr }) => {
    if (hErr) console.error('[board] status_history insert:', hErr);
  });
  return true;
}
```

```javascript
async function saveAcceptance() {
  const sb = window._sb;
  const ctx = window._boardCtx;
  const btn = document.getElementById('btnAcceptSave');
  // lockButton from ui.js is module-exported; use inline disable pattern instead
  btn.disabled = true;
  const origText = btn.textContent;
  btn.textContent = 'Сохранение...';
  try {
    const payload = gatherAcceptanceForm();
    // Step 1: INSERT acceptance_acts
    const { error: insertErr } = await sb
      .from('acceptance_acts')
      .insert({
        calc_id: _acceptCalcId,
        studio_id: ctx.studioId,
        ...payload,
      });
    // Handle duplicate key (23505) as success — record already exists (Pitfall 2)
    if (insertErr) {
      const isDuplicate = insertErr.code === '23505' || (insertErr.message && insertErr.message.includes('duplicate'));
      if (!isDuplicate) {
        console.error('[pipeline] acceptance insert:', insertErr);
        showToast('error', 'Ошибка сохранения акта приёмки');
        return;
      }
    }
    // Step 2: PATCH status via board's updateStatus
    const ok = await window._updateStatus(_acceptCalcId, 'accepted');
    if (ok) {
      document.getElementById('modalAccept').classList.remove('active');
      showToast('success', 'Авто принято');
      window._loadBoard();
    }
  } finally {
    btn.disabled = false;
    btn.textContent = origText;
  }
}
```
